	BounceCount 4
	+DOOMBOUNCE
    +NOBLOCKMAP
	+MISSILE
    +NOTELEPORT
    +DONTSPLASH
    +MOVEWITHSECTOR
    +CLIENTSIDEONLY
	+THRUACTORS
    Scale 0.4
    States
    {
    Spawn:
		TNT1 A 0
		TNT1 A 0 Thing_ChangeTID(0,394)

		wait
		SPIC HGFEDCBA 2
		wait

		TNT1 A 0
		TNT1 A 0 A_Jump(128,2)
		SPIC A 401
		Stop
		SPIC E 1
		SPIC E 401
        Stop
    }
}
